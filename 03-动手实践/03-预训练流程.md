# 预训练流程

## 学习目标
学完本节后，你将能够：
- 理解 LLM 预训练的完整流程
- 掌握数据准备和预处理的方法
- 了解训练过程中的关键技术
- 能够搭建和运行简单的预训练流程

## 核心知识点

### 1. 预训练数据准备

#### 1.1 数据来源
- **网页数据**：Common Crawl、维基百科
- **书籍数据**：Project Gutenberg、古登堡计划
- **代码数据**：GitHub、Stack Overflow
- **学术论文**：arXiv、PubMed
- **新闻文章**：新闻网站、通讯社

#### 1.2 数据清洗流程
1. **去重**：移除重复内容，减少偏差
2. **过滤**：移除低质量、有害内容
3. **格式化**：统一文本格式，处理特殊字符
4. **分词**：转换为 token 序列

#### 1.3 数据混合策略
- **比例分配**：不同类型数据的比例
- **课程学习**：从简单到复杂的数据
- **动态采样**：根据训练进度调整采样

### 2. 训练基础设施

#### 2.1 硬件需求
- **GPU**：A100/H100 等高端 GPU
- **内存**：大量系统内存用于数据加载
- **存储**：高速 SSD 用于数据读取
- **网络**：高速互连（NVLink、InfiniBand）

#### 2.2 分布式训练
- **数据并行**：不同 GPU 处理不同数据
- **模型并行**：不同 GPU 处理不同层
- **流水线并行**：不同 GPU 处理不同阶段
- **混合并行**：结合多种并行策略

#### 2.3 优化技术
- **混合精度训练**：FP16/BF16 加速
- **梯度累积**：等效增大批大小
- **激活检查点**：减少内存占用
- **优化器状态分片**：ZeRO 优化

### 3. 训练过程

#### 3.1 训练循环
```
for epoch in range(num_epochs):
    for batch in dataloader:
        # 前向传播
        outputs = model(batch)
        loss = loss_fn(outputs, targets)
        
        # 反向传播
        loss.backward()
        
        # 梯度裁剪
        clip_grad_norm_(model.parameters(), max_norm)
        
        # 优化器更新
        optimizer.step()
        optimizer.zero_grad()
        
        # 学习率调度
        scheduler.step()
        
        # 日志记录
        log_metrics(loss, lr, step)
```

#### 3.2 学习率调度
- **预热阶段**：线性增加学习率
- **稳定阶段**：保持或缓慢衰减
- **衰减阶段**：余弦衰减或线性衰减

#### 3.3 监控与调试
- **损失曲线**：监控训练损失
- **梯度统计**：检查梯度爆炸/消失
- **激活分布**：监控模型激活
- **显存使用**：避免 OOM

### 4. 预训练任务

#### 4.1 因果语言建模（CLM）
- **目标**：预测下一个 token
- **损失**：交叉熵损失
- **适用**：GPT 系列模型

#### 4.2 掩码语言建模（MLM）
- **目标**：预测被掩码的 token
- **掩码比例**：通常 15%
- **适用**：BERT 系列模型

#### 4.3 去噪自编码（DAE）
- **目标**：重建被破坏的输入
- **破坏方式**：掩码、删除、替换
- **适用**：T5、BART 等模型

## 代码示例

### 完整预训练流程

```python
import torch
import torch.nn as nn
from torch.utils.data import Dataset, DataLoader
from transformers import get_linear_schedule_with_warmup
import math
from tqdm import tqdm

class TextDataset(Dataset):
    """文本数据集"""
    
    def __init__(self, texts, tokenizer, max_length=512):
        self.tokenizer = tokenizer
        self.max_length = max_length
        self.examples = []
        
        # 处理所有文本
        for text in texts:
            # 编码
            encoded = tokenizer.encode(text)
            
            # 分割为固定长度的序列
            for i in range(0, len(encoded) - max_length, max_length):
                chunk = encoded[i:i+max_length]
                if len(chunk) == max_length:
                    self.examples.append(chunk)
    
    def __len__(self):
        return len(self.examples)
    
    def __getitem__(self, idx):
        return torch.tensor(self.examples[idx], dtype=torch.long)

class DataCollator:
    """数据整理器"""
    
    def __init__(self, pad_token_id=0):
        self.pad_token_id = pad_token_id
    
    def __call__(self, batch):
        # 找到最大长度
        max_len = max(x.size(0) for x in batch)
        
        # 填充
        input_ids = []
        labels = []
        
        for x in batch:
            # 输入：序列去掉最后一个 token
            input_seq = x[:-1]
            # 目标：序列去掉第一个 token（预测下一个）
            label_seq = x[1:]
            
            # 填充
            pad_len = max_len - 1 - input_seq.size(0)
            input_seq = torch.cat([input_seq, torch.full((pad_len,), self.pad_token_id)])
            label_seq = torch.cat([label_seq, torch.full((pad_len,), -100)])  # -100 忽略损失
            
            input_ids.append(input_seq)
            labels.append(label_seq)
        
        return {
            'input_ids': torch.stack(input_ids),
            'labels': torch.stack(labels)
        }

class PretrainingTrainer:
    """预训练训练器"""
    
    def __init__(self, model, train_dataset, val_dataset=None, 
                 batch_size=8, learning_rate=1e-4, weight_decay=0.01,
                 warmup_steps=1000, max_steps=100000, gradient_accumulation_steps=1,
                 fp16=False):
        
        self.model = model
        self.train_dataset = train_dataset
        self.val_dataset = val_dataset
        self.batch_size = batch_size
        self.learning_rate = learning_rate
        self.weight_decay = weight_decay
        self.warmup_steps = warmup_steps
        self.max_steps = max_steps
        self.gradient_accumulation_steps = gradient_accumulation_steps
        self.fp16 = fp16
        
        # 数据整理器
        self.data_collator = DataCollator()
        
        # 数据加载器
        self.train_dataloader = DataLoader(
            train_dataset,
            batch_size=batch_size,
            shuffle=True,
            collate_fn=self.data_collator,
            num_workers=4,
            pin_memory=True
        )
        
        # 优化器
        self.optimizer = torch.optim.AdamW(
            model.parameters(),
            lr=learning_rate,
            weight_decay=weight_decay,
            betas=(0.9, 0.95)
        )
        
        # 学习率调度器
        self.total_steps = min(max_steps, len(self.train_dataloader))
        self.scheduler = get_linear_schedule_with_warmup(
            self.optimizer,
            num_warmup_steps=warmup_steps,
            num_training_steps=self.total_steps
        )
        
        # 混合精度
        self.scaler = torch.cuda.amp.GradScaler() if fp16 else None
        
        # 设备
        self.device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
        self.model.to(self.device)
        
        # 训练状态
        self.global_step = 0
        self.best_loss = float('inf')
    
    def train_step(self, batch):
        """单步训练"""
        input_ids = batch['input_ids'].to(self.device)
        labels = batch['labels'].to(self.device)
        
        # 混合精度训练
        if self.fp16:
            with torch.cuda.amp.autocast():
                outputs = self.model(input_ids, labels=labels)
                loss = outputs.loss
        else:
            outputs = self.model(input_ids, labels=labels)
            loss = outputs.loss
        
        # 梯度累积
        loss = loss / self.gradient_accumulation_steps
        
        # 反向传播
        if self.fp16:
            self.scaler.scale(loss).backward()
        else:
            loss.backward()
        
        return loss.item() * self.gradient_accumulation_steps
    
    def train(self):
        """训练主循环"""
        self.model.train()
        
        progress_bar = tqdm(total=self.max_steps, desc="Training")
        
        for epoch in range(1000):  # 大的 epoch 数，实际由 max_steps 控制
            for step, batch in enumerate(self.train_dataloader):
                # 训练步
                loss = self.train_step(batch)
                
                # 梯度累积
                if (step + 1) % self.gradient_accumulation_steps == 0:
                    # 梯度裁剪
                    if self.fp16:
                        self.scaler.unscale_(self.optimizer)
                    torch.nn.utils.clip_grad_norm_(self.model.parameters(), 1.0)
                    
                    # 优化器更新
                    if self.fp16:
                        self.scaler.step(self.optimizer)
                        self.scaler.update()
                    else:
                        self.optimizer.step()
                    
                    self.scheduler.step()
                    self.optimizer.zero_grad()
                    
                    self.global_step += 1
                    
                    # 更新进度条
                    progress_bar.update(1)
                    progress_bar.set_postfix({
                        'loss': f'{loss:.4f}',
                        'lr': f'{self.scheduler.get_last_lr()[0]:.2e}',
                        'step': self.global_step
                    })
                    
                    # 定期保存
                    if self.global_step % 1000 == 0:
                        self.save_checkpoint()
                    
                    # 检查停止条件
                    if self.global_step >= self.max_steps:
                        print(f"达到最大步数 {self.max_steps}，训练结束")
                        return
                    
                    # 验证
                    if self.val_dataset and self.global_step % 500 == 0:
                        val_loss = self.validate()
                        print(f"\n验证损失: {val_loss:.4f}")
                        
                        if val_loss < self.best_loss:
                            self.best_loss = val_loss
                            self.save_checkpoint(is_best=True)
                        
                        self.model.train()
    
    def validate(self):
        """验证"""
        self.model.eval()
        
        val_dataloader = DataLoader(
            self.val_dataset,
            batch_size=self.batch_size,
            shuffle=False,
            collate_fn=self.data_collator
        )
        
        total_loss = 0
        total_steps = 0
        
        with torch.no_grad():
            for batch in tqdm(val_dataloader, desc="Validating"):
                input_ids = batch['input_ids'].to(self.device)
                labels = batch['labels'].to(self.device)
                
                outputs = self.model(input_ids, labels=labels)
                loss = outputs.loss
                
                total_loss += loss.item()
                total_steps += 1
        
        return total_loss / total_steps
    
    def save_checkpoint(self, is_best=False):
        """保存检查点"""
        checkpoint = {
            'global_step': self.global_step,
            'model_state_dict': self.model.state_dict(),
            'optimizer_state_dict': self.optimizer.state_dict(),
            'scheduler_state_dict': self.scheduler.state_dict(),
            'loss': self.best_loss,
        }
        
        if self.scaler:
            checkpoint['scaler_state_dict'] = self.scaler.state_dict()
        
        filename = f'checkpoint-{self.global_step}.pt'
        if is_best:
            filename = 'best-checkpoint.pt'
        
        torch.save(checkpoint, filename)
        print(f"保存检查点: {filename}")

# 使用示例
if __name__ == "__main__":
    # 简化的模型和 tokenizer（实际需要完整的实现）
    class SimpleModel(nn.Module):
        def __init__(self, vocab_size, d_model=256):
            super().__init__()
            self.embedding = nn.Embedding(vocab_size, d_model)
            self.transformer = nn.TransformerEncoderLayer(d_model, nhead=4)
            self.lm_head = nn.Linear(d_model, vocab_size)
        
        def forward(self, input_ids, labels=None):
            x = self.embedding(input_ids)
            x = self.transformer(x)
            logits = self.lm_head(x)
            
            loss = None
            if labels is not None:
                loss = nn.CrossEntropyLoss()(logits.view(-1, logits.size(-1)), labels.view(-1))
            
            return type('Outputs', (), {'logits': logits, 'loss': loss})()
    
    class SimpleTokenizer:
        def __init__(self):
            self.vocab = {chr(i): i for i in range(256)}
            self.vocab_size = 256
        
        def encode(self, text):
            return [self.vocab.get(c, 0) for c in text]
        
        def decode(self, ids):
            return ''.join([chr(i) if i < 256 else '?' for i in ids])
    
    # 准备数据
    sample_texts = [
        "机器学习是人工智能的一个重要分支。",
        "深度学习是机器学习的一个子集。",
        "大语言模型是深度学习的重要应用。"
    ] * 100  # 重复以增加数据量
    
    tokenizer = SimpleTokenizer()
    dataset = TextDataset(sample_texts, tokenizer, max_length=128)
    
    print(f"数据集大小: {len(dataset)}")
    print(f"单个样本长度: {dataset[0].size(0)}")
    
    # 创建模型
    model = SimpleModel(vocab_size=tokenizer.vocab_size)
    
    # 创建训练器
    trainer = PretrainingTrainer(
        model=model,
        train_dataset=dataset,
        batch_size=4,
        learning_rate=1e-4,
        max_steps=100  # 短训练用于演示
    )
    
    # 开始训练
    trainer.train()
```
