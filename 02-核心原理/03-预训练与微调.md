# 预训练与微调

## 学习目标
学完本节后，你将能够：

- 掌握不同预训练任务和微调策略

- 能够实现简单的微调流程

## 核心知识点

### 1. 预训练（Pre-training）

#### 1.1 定义与目标
预训练是在大规模无标注文本数据上训练模型，学习通用的语言表示和知识。

**核心思想**：
- 先在大规模数据上学习通用语言模式
- 再在特定任务上进行微调
- 迁移学习在 NLP 领域的成功实践

#### 1.2 预训练任务类型

**1.2.1 因果语言建模（Causal LM）**
- **目标**：预测下一个词（Next Token Prediction）
- **代表模型**：GPT 系列
- **特点**：单向，自回归生成

**1.2.2 掩码语言建模（Masked LM）**
- **目标**：预测被掩码的词
- **代表模型**：BERT 系列
- **特点**：双向上下文理解

**1.2.3 序列到序列建模（Seq2Seq）**
- **目标**：输入序列 → 输出序列
- **代表模型**：T5、BART
- **特点**：编码器-解码器结构

#### 1.3 预训练数据
- **来源**：网页、书籍、代码、维基百科等
- **规模**：TB 级别文本数据
- **处理**：清洗、去重、格式化

### 2. 微调（Fine-tuning）

#### 2.1 定义与流程
微调是在预训练模型基础上，使用特定任务数据进行参数调整。

**标准流程**：
1. 加载预训练模型
2. 准备任务特定数据
3. 调整模型输出层
4. 在任务数据上训练

#### 2.2 微调类型

**2.2.1 全参数微调**
- **方法**：更新所有模型参数
- **优势**：性能最大化
- **劣势**：计算成本高，需要大量存储

**2.2.2 参数高效微调（PEFT）**
- **方法**：只更新部分参数
- **代表技术**：
  - **LoRA**：低秩适配器
  - **Adapter**：插入小型适配模块
  - **Prompt Tuning**：优化提示参数
- **优势**：节省计算和存储资源

#### 2.3 指令微调（Instruction Tuning）
- **目标**：让模型学会遵循人类指令
- **数据**：指令-输出对
- **代表**：InstructGPT、ChatGPT

### 3. 人类反馈强化学习（RLHF）

#### 3.1 核心思想
通过人类反馈训练奖励模型，使用强化学习优化模型输出。

#### 3.2 三阶段流程
1. **监督微调（SFT）**：在演示数据上微调
2. **奖励模型训练**：学习人类偏好
3. **强化学习优化**：使用 PPO 算法优化

#### 3.3 优势
- 对齐人类价值观
- 提高输出质量和安全性
- 减少有害内容生成

### 4. 预训练 vs 微调对比

| 维度 | 预训练 | 微调 |
|------|--------|------|
| **目标** | 学习通用语言表示 | 适应特定任务 |
| **数据** | 大规模无标注文本 | 小规模标注数据 |
| **计算成本** | 极高（数百万 GPU 小时） | 较低（数小时到数天） |
| **模型状态** | 通用但不专门化 | 任务专门化 |
| **应用场景** | 模型初始化 | 实际部署 |

import torch
import torch.nn as nn
import torch.nn.functional as F
from transformers import AutoTokenizer, AutoModelForCausalLM

class MaskedLanguageModeling(nn.Module):
    """掩码语言建模（BERT 预训练任务）"""
    
    def __init__(self, vocab_size, d_model=768):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size, d_model)
        self.transformer = nn.TransformerEncoderLayer(d_model, nhead=12)
        self.mlm_head = nn.Linear(d_model, vocab_size)
    
    def forward(self, input_ids, attention_mask=None):
        # 获取嵌入
        embeddings = self.embedding(input_ids)
        
        # Transformer 编码
        if attention_mask is not None:
            # 创建注意力掩码
            attention_mask = attention_mask.bool()
        
        # 这里简化处理，实际需要完整的 Transformer
        output = embeddings  # 简化
        
        # 预测掩码位置
        logits = self.mlm_head(output)
        return logits

class CausalLanguageModeling(nn.Module):
    """因果语言建模（GPT 预训练任务）"""
    
    def __init__(self, vocab_size, d_model=768):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size, d_model)
        self.transformer = nn.TransformerEncoderLayer(d_model, nhead=12)
        self.lm_head = nn.Linear(d_model, vocab_size)
    
    def create_causal_mask(self, seq_len):
        """创建因果掩码"""
        mask = torch.triu(torch.ones(seq_len, seq_len), diagonal=1).bool()
        return ~mask
    
    def forward(self, input_ids):
        # 获取嵌入
        embeddings = self.embedding(input_ids)
        
        # 创建因果掩码
        seq_len = input_ids.size(1)
        causal_mask = self.create_causal_mask(seq_len).to(input_ids.device)
        
        # Transformer 编码（简化）
        output = embeddings  # 简化
        
        # 预测下一个词
        logits = self.lm_head(output)
        return logits

# 测试预训练任务
vocab_size = 10000
model = CausalLanguageModeling(vocab_size)

batch_size, seq_len = 2, 10
input_ids = torch.randint(0, vocab_size, (batch_size, seq_len))
logits = model(input_ids)

print(f"输入形状: {input_ids.shape}")
print(f"输出 logits 形状: {logits.shape}")

### LoRA 微调实现

```python
import torch
import torch.nn as nn

class LoRALayer(nn.Module):
    """LoRA 适配器层"""
    
    def __init__(self, original_layer, rank=8, alpha=16):
        super().__init__()
        self.original_layer = original_layer
        self.rank = rank
        self.alpha = alpha
        
        # 冻结原始层参数
        for param in self.original_layer.parameters():
            param.requires_grad = False
        
        # 创建 LoRA 参数
        in_features = original_layer.in_features
        out_features = original_layer.out_features
        
        self.lora_A = nn.Parameter(torch.randn(in_features, rank))
        self.lora_B = nn.Parameter(torch.zeros(rank, out_features))
        self.scaling = alpha / rank
    
    def forward(self, x):
        # 原始层输出
        original_output = self.original_layer(x)
        
        # LoRA 路径
        lora_output = x @ self.lora_A @ self.lora_B * self.scaling
        
        # 合并输出
        return original_output + lora_output

class LoRALinear(nn.Module):
    """带 LoRA 的线性层"""
    
    def __init__(self, in_features, out_features, rank=8, alpha=16):
        super().__init__()
        self.original_linear = nn.Linear(in_features, out_features)
        self.lora = LoRALayer(self.original_linear, rank, alpha)
    
    def forward(self, x):
        return self.lora(x)

# 测试 LoRA
input_dim, output_dim = 512, 256
x = torch.randn(2, 10, input_dim)

# 普通线性层
linear = nn.Linear(input_dim, output_dim)
print(f"普通线性层参数量: {sum(p.numel() for p in linear.parameters()):,}")

# LoRA 线性层
lora_linear = LoRALinear(input_dim, output_dim, rank=8)
lora_params = sum(p.numel() for p in lora_linear.parameters() if p.requires_grad)
total_params = sum(p.numel() for p in lora_linear.parameters())
print(f"LoRA 线性层总参数量: {total_params:,}")
print(f"LoRA 可训练参数量: {lora_params:,} ({lora_params/total_params*100:.1f}%)")

# 前向传播
output = lora_linear(x)
print(f"输出形状: {output.shape}")
```

### 指令微调数据准备

```python
class InstructionDataset:
    """指令微调数据集"""
    
    def __init__(self, tokenizer, max_length=512):
        self.tokenizer = tokenizer
        self.max_length = max_length
        
        # 示例指令数据
        self.data = [
            {
                "instruction": "解释什么是机器学习",
                "input": "",
                "output": "机器学习是人工智能的一个分支，它使计算机能够从数据中学习，而无需进行明确编程。"
            },
            {
                "instruction": "将以下句子翻译成英文",
                "input": "今天天气很好",
                "output": "The weather is very nice today."
            },
            {
                "instruction": "写一首关于春天的诗",
                "input": "",
                "output": "春风拂面柳丝长，\n百花齐放满园香。\n燕子归来寻旧巢，\n人间处处好风光。"
            }
        ]
    
    def format_instruction(self, instruction, input_text="", output=""):
        """格式化指令"""
        if input_text:
            prompt = f"指令：{instruction}\n输入：{input_text}\n输出："
        else:
            prompt = f"指令：{instruction}\n输出："
        return prompt + output
    
    def tokenize(self, example):
        """分词处理"""
        text = self.format_instruction(
            example["instruction"],
            example["input"],
            example["output"]
        )
        
        # 编码
        encoding = self.tokenizer(
            text,
            truncation=True,
            max_length=self.max_length,
            padding="max_length",
            return_tensors="pt"
        )
        
        # 创建标签（输入部分设为 -100，不计算损失）
        labels = encoding["input_ids"].clone()
        # 这里简化处理，实际需要更复杂的逻辑
        
        return {
            "input_ids": encoding["input_ids"].squeeze(),
            "attention_mask": encoding["attention_mask"].squeeze(),
            "labels": labels.squeeze()
        }
    
    def __len__(self):
        return len(self.data)
    
    def __getitem__(self, idx):
        return self.tokenize(self.data[idx])

# 使用示例
from transformers import AutoTokenizer
tokenizer = AutoTokenizer.from_pretrained("bert-base-uncased")
dataset = InstructionDataset(tokenizer)
print(f"数据集大小: {len(dataset)}")
print(f"第一个样本: {dataset[0]}")
```
