# 注意力机制

## 学习目标
学完本节后，你将能够：

## 核心知识点

### 1. 注意力机制基础

#### 1.1 核心思想
注意力机制允许模型在处理序列时，动态地关注输入序列的不同部分，而不是对所有部分一视同仁。

**直观理解**：
- 人类阅读时，会重点关注关键词
- 注意力机制模拟这种"选择性关注"

#### 1.2 数学表达
```
Attention(Q, K, V) = softmax(QK^T / √d_k) V
```

**组件解释**：
- **Q（Query）**：查询向量，表示"我在找什么"
- **K（Key）**：键向量，表示"我有什么信息"
- **V（Value）**：值向量，表示"我的实际内容"
- **d_k**：键向量的维度，用于缩放

#### 1.3 计算步骤
1. 计算相似度：Q 与 K 的点积
2. 缩放：除以 √d_k 防止梯度消失
3. 归一化：softmax 得到注意力权重
4. 加权求和：权重 × V

### 2. 注意力机制类型

#### 2.1 自注意力（Self-Attention）
- **定义**：Q、K、V 都来自同一输入序列
- **应用**：Transformer 编码器、解码器自注意力层
- **优势**：捕获序列内部依赖关系

#### 2.2 交叉注意力（Cross-Attention）
- **定义**：Q 来自解码器，K、V 来自编码器
- **应用**：Transformer 解码器中的编码器-解码器注意力
- **作用**：让解码器关注编码器的输出

#### 2.3 掩码注意力（Masked Attention）
- **定义**：防止当前位置关注未来位置
- **实现**：将未来位置设为 -inf
- **应用**：解码器的自注意力层（训练时）

### 3. 多头注意力机制

#### 3.1 动机
- 单个注意力头可能只关注一种模式
- 多头允许模型同时关注不同方面的信息

#### 3.2 实现细节
```
MultiHead(Q, K, V) = Concat(head_1, ..., head_h) W^O
其中 head_i = Attention(QW_i^Q, KW_i^K, VW_i^V)
```

#### 3.3 优势
- **并行性**：多个头可以同时计算
- **多样性**：不同头关注不同特征
- **表达能力**：增强模型的表示能力

### 4. 注意力变体

#### 4.1 稀疏注意力（Sparse Attention）
- **问题**：标准注意力是 O(n²) 复杂度
- **解决方案**：只计算部分位置的注意力
- **应用**：长序列处理（如 Longformer、BigBird）

#### 4.2 线性注意力（Linear Attention）
- **原理**：使用核函数近似注意力计算
- **复杂度**：O(n) 而非 O(n²)
- **限制**：可能损失一些表达能力

#### 4.3 闪光注意力（Flash Attention）
- **目标**：减少 GPU 内存访问，加速训练
- **特点**：IO 感知的精确注意力计算
- **优势**：速度更快，内存占用更少

## 代码示例

### 不同注意力机制实现

```python
import torch
import torch.nn as nn
import torch.nn.functional as F
import math

class SelfAttention(nn.Module):
    """自注意力机制"""
    
    def __init__(self, d_model, dropout=0.1):
        super().__init__()
        self.d_model = d_model
        self.W_q = nn.Linear(d_model, d_model)
        self.W_k = nn.Linear(d_model, d_model)
        self.W_v = nn.Linear(d_model, d_model)
        self.dropout = nn.Dropout(dropout)
    
    def forward(self, x, mask=None):
        batch_size, seq_len, d_model = x.shape
        
        # 计算 Q, K, V
        Q = self.W_q(x)
        K = self.W_k(x)
        V = self.W_v(x)
        
        # 计算注意力分数
        scores = torch.matmul(Q, K.transpose(-2, -1)) / math.sqrt(d_model)
        
        # 应用掩码
        if mask is not None:
            scores = scores.masked_fill(mask == 0, -1e9)
        
        # Softmax 归一化
        attention_weights = F.softmax(scores, dim=-1)
        attention_weights = self.dropout(attention_weights)
        
        # 加权求和
        output = torch.matmul(attention_weights, V)
        
        return output, attention_weights

class CrossAttention(nn.Module):
    """交叉注意力机制"""
    
    def __init__(self, d_model, dropout=0.1):
        super().__init__()
        self.d_model = d_model
        self.W_q = nn.Linear(d_model, d_model)
        self.W_k = nn.Linear(d_model, d_model)
        self.W_v = nn.Linear(d_model, d_model)
        self.dropout = nn.Dropout(dropout)
    
    def forward(self, query, key_value, mask=None):
        # query: [batch, seq_len_q, d_model]
        # key_value: [batch, seq_len_kv, d_model]
        
        Q = self.W_q(query)
        K = self.W_k(key_value)
        V = self.W_v(key_value)
        
        scores = torch.matmul(Q, K.transpose(-2, -1)) / math.sqrt(self.d_model)
        
        if mask is not None:
            scores = scores.masked_fill(mask == 0, -1e9)
        
        attention_weights = F.softmax(scores, dim=-1)
        attention_weights = self.dropout(attention_weights)
        
        output = torch.matmul(attention_weights, V)
        return output, attention_weights

class MaskedSelfAttention(nn.Module):
    """掩码自注意力（用于解码器）"""
    
    def __init__(self, d_model, dropout=0.1):
        super().__init__()
        self.d_model = d_model
        self.W_q = nn.Linear(d_model, d_model)
        self.W_k = nn.Linear(d_model, d_model)
        self.W_v = nn.Linear(d_model, d_model)
        self.dropout = nn.Dropout(dropout)
    
    def create_causal_mask(self, seq_len):
        """创建因果掩码（下三角）"""
        mask = torch.triu(torch.ones(seq_len, seq_len), diagonal=1).bool()
        return ~mask  # 下三角为 True，上三角为 False
    
    def forward(self, x):
        batch_size, seq_len, d_model = x.shape
        
        Q = self.W_q(x)
        K = self.W_k(x)
        V = self.W_v(x)
        
        scores = torch.matmul(Q, K.transpose(-2, -1)) / math.sqrt(d_model)
        
        # 应用因果掩码
        causal_mask = self.create_causal_mask(seq_len).to(x.device)
        scores = scores.masked_fill(~causal_mask, -1e9)
        
        attention_weights = F.softmax(scores, dim=-1)
        attention_weights = self.dropout(attention_weights)
        
        output = torch.matmul(attention_weights, V)
        return output, attention_weights

# 测试不同的注意力机制
batch_size, seq_len, d_model = 2, 10, 64

# 自注意力
self_attn = SelfAttention(d_model)
x = torch.randn(batch_size, seq_len, d_model)
output, weights = self_attn(x)
print(f"自注意力输出形状: {output.shape}")
print(f"注意力权重形状: {weights.shape}")

# 交叉注意力
cross_attn = CrossAttention(d_model)
query = torch.randn(batch_size, 5, d_model)
key_value = torch.randn(batch_size, 10, d_model)
output, weights = cross_attn(query, key_value)
print(f"交叉注意力输出形状: {output.shape}")

# 掩码自注意力
masked_attn = MaskedSelfAttention(d_model)
output, weights = masked_attn(x)
print(f"掩码自注意力输出形状: {output.shape}")
```

### 注意力权重可视化

```python
import matplotlib.pyplot as plt
import seaborn as sns

def visualize_attention(attention_weights, tokens=None, title="注意力权重"):
    """可视化注意力权重矩阵"""
    plt.figure(figsize=(10, 8))
    
    # 取第一个样本
    weights = attention_weights[0].detach().cpu().numpy()
    
    # 如果有多个头，取第一个头
    if len(weights.shape) == 3:
        weights = weights[0]
    
    sns.heatmap(weights, cmap='YlOrRd', annot=False, 
                xticklabels=tokens, yticklabels=tokens)
    plt.xlabel('Key 位置')
    plt.ylabel('Query 位置')
    plt.title(title)
    plt.tight_layout()
    plt.savefig('attention_weights.png')
    plt.show()

# 使用示例
batch_size, seq_len, d_model = 1, 5, 32
self_attn = SelfAttention(d_model)
x = torch.randn(batch_size, seq_len, d_model)
_, weights = self_attn(x)

tokens = ['<start>', 'the', 'cat', 'sat', '<end>']
visualize_attention(weights, tokens, "自注意力权重可视化")
```

### 稀疏注意力实现

```python
class SparseAttention(nn.Module):
    """稀疏注意力（局部注意力）"""
    
    def __init__(self, d_model, window_size=3, dropout=0.1):
        super().__init__()
        self.d_model = d_model
        self.window_size = window_size
        self.W_q = nn.Linear(d_model, d_model)
        self.W_k = nn.Linear(d_model, d_model)
        self.W_v = nn.Linear(d_model, d_model)
        self.dropout = nn.Dropout(dropout)
    
    def create_local_mask(self, seq_len, window_size):
        """创建局部注意力掩码"""
        mask = torch.zeros(seq_len, seq_len)
        for i in range(seq_len):
            start = max(0, i - window_size)
            end = min(seq_len, i + window_size + 1)
            mask[i, start:end] = 1
        return mask.bool()
    
    def forward(self, x):
        batch_size, seq_len, d_model = x.shape
        
        Q = self.W_q(x)
        K = self.W_k(x)
        V = self.W_v(x)
        
        scores = torch.matmul(Q, K.transpose(-2, -1)) / math.sqrt(d_model)
        
        # 应用局部注意力掩码
        local_mask = self.create_local_mask(seq_len, self.window_size).to(x.device)
        scores = scores.masked_fill(~local_mask, -1e9)
        
        attention_weights = F.softmax(scores, dim=-1)
        attention_weights = self.dropout(attention_weights)
        
        output = torch.matmul(attention_weights, V)
        return output, attention_weights

# 测试稀疏注意力
sparse_attn = SparseAttention(d_model=64, window_size=2)
x = torch.randn(2, 10, 64)
output, weights = sparse_attn(x)
print(f"稀疏注意力输出形状: {output.shape}")
```
