# 实战项目一：对话机器人

> 从零搭建一个功能完整的对话机器人，涵盖 API 调用、上下文管理、流式输出等核心技能。

## 项目目标

构建一个具备以下能力的对话机器人：

1. **基础对话** — 支持多轮对话，记住上下文
2. **角色设定** — 可自定义系统提示词（System Prompt）
3. **流式输出** — 打字机效果，实时显示生成内容
4. **历史管理** — 支持保存、加载、清空对话历史
5. **错误处理** — 优雅处理 API 异常和网络问题

## 技术栈选择

## 核心代码实现

### 1. 基础对话类

```python
import os
from openai import OpenAI
from dotenv import load_dotenv

load_dotenv()

class ChatBot:
    def __init__(self, model="gpt-4", system_prompt="你是一个有帮助的AI助手。"):
        self.client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))
        self.model = model
        self.messages = [
            {"role": "system", "content": system_prompt}
        ]
    
    def chat(self, user_input: str) -> str:
        """发送消息并获取回复"""
        self.messages.append({"role": "user", "content": user_input})
        
        response = self.client.chat.completions.create(
            model=self.model,
            messages=self.messages
        )
        
        reply = response.choices[0].message.content
        self.messages.append({"role": "assistant", "content": reply})
        return reply
    
    def reset(self):
        """重置对话历史"""
        system_msg = self.messages[0]  # 保留系统提示词
        self.messages = [system_msg]
    
    def get_history(self):
        """获取对话历史"""
        return self.messages
```

### 2. 流式输出实现

```python
def chat_stream(self, user_input: str):
    """流式输出对话"""
    self.messages.append({"role": "user", "content": user_input})
    
    response = self.client.chat.completions.create(
        model=self.model,
        messages=self.messages,
        stream=True  # 开启流式
    )
    
    full_reply = ""
    print("🤖 ", end="", flush=True)
    
    for chunk in response:
        if chunk.choices[0].delta.content:
            content = chunk.choices[0].delta.content
            print(content, end="", flush=True)
            full_reply += content
    
    print()  # 换行
    self.messages.append({"role": "assistant", "content": full_reply})
    return full_reply
```

### 3. 历史记录持久化

```python
import json
from datetime import datetime

class PersistentChatBot(ChatBot):
    def __init__(self, *args, history_dir="chat_history", **kwargs):
        super().__init__(*args, **kwargs)
        self.history_dir = history_dir
        os.makedirs(history_dir, exist_ok=True)
    
    def save_session(self, session_id: str = None):
        """保存当前会话"""
        if not session_id:
            session_id = datetime.now().strftime("%Y%m%d_%H%M%S")
        
        filepath = os.path.join(self.history_dir, f"{session_id}.json")
        with open(filepath, "w", encoding="utf-8") as f:
            json.dump({
                "model": self.model,
                "messages": self.messages,
                "timestamp": datetime.now().isoformat()
            }, f, ensure_ascii=False, indent=2)
        
        return filepath
    
    def load_session(self, session_id: str):
        """加载历史会话"""
        filepath = os.path.join(self.history_dir, f"{session_id}.json")
        with open(filepath, "r", encoding="utf-8") as f:
            data = json.load(f)
        
        self.model = data["model"]
        self.messages = data["messages"]
        return len(self.messages) - 1  # 返回消息数量（不含system）
```

## 命令行界面示例

```python
# main.py
def main():
    bot = PersistentChatBot(
        model="gpt-4",
        system_prompt="你是一个专业、友善的技术助手。"
    )
    
    print("=" * 50)
    print("🤖 对话机器人已启动")
    print("输入 'quit' 退出，'reset' 重置，'save' 保存")
    print("=" * 50)
    
    while True:
        try:
            user_input = input("\n👤 你: ")
            
            if user_input.lower() == 'quit':
                bot.save_session()
                print("👋 再见！")
                break
            elif user_input.lower() == 'reset':
                bot.reset()
                print("✅ 对话已重置")
                continue
            elif user_input.lower() == 'save':
                path = bot.save_session()
                print(f"💾 已保存到: {path}")
                continue
            
            bot.chat_stream(user_input)
            
        except KeyboardInterrupt:
            bot.save_session()
            print("\n👋 已保存并退出")
            break
        except Exception as e:
            print(f"❌ 错误: {e}")

if __name__ == "__main__":
    main()
```

### 2. 多角色对话

```python
class MultiRoleChat:
    def __init__(self):
        self.roles = {
            "assistant": "你是一个有帮助的助手。",
            "translator": "你是一个专业的翻译，将用户的内容翻译成英文。",
            "coder": "你是一个资深程序员，精通 Python 和 JavaScript。",
        }
    
    def switch_role(self, role_name: str):
        """切换角色"""
        if role_name in self.roles:
            self.messages[0]["content"] = self.roles[role_name]
            return True
        return False
```

### 3. RAG 增强对话

结合检索增强生成，让机器人回答基于特定知识库的问题。

```python
# 简化版 RAG
def rag_chat(self, user_input, knowledge_base):
    # 1. 检索相关文档
    relevant_docs = knowledge_base.search(user_input, top_k=3)
    
    # 2. 构建增强提示词
    context = "\n".join(relevant_docs)
    enhanced_input = f"""基于以下上下文回答问题：

上下文：
{context}

问题：{user_input}"""
    
    # 3. 发送请求
    return self.chat(enhanced_input)
```
